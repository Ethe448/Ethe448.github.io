def encrypt(data, key):
    #   assert len(data) <= len(key)
    result = ""
    for i in range(len(data)):
        if data[i].isupper():
            result += chr((ord(data[i]) - ord('A') + key[i]) % 26 + ord('A'))

            result += chr((ord(data[i]) - ord('a') + key[i]) % 26 + ord('a'))
